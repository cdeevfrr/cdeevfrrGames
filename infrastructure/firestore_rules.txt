rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Only allow users to access documents in their own 'users/{uid}/nodes' sub-collection
    match /users/{userId}/{x=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    // Callable functions often need to write to other collections, 
    // or run complex logic, but we only need to define these rules for direct client access.
    // The Admin SDK (used by the functions) bypasses these rules entirely.
  }
}
